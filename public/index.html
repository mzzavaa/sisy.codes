SISY API

A clean REST API to search experts, retrieve profiles, and surface opportunities. Built for organizers, event tools, and partner integrations.

sisy-api · example.js 
// Search SISY experts by topic and location
const response = await fetch(
  'https://api.sisy.codes/v1/experts?topic=AI&location=Berlin',
  {
    headers: {
      'Authorization': `Bearer ${API_KEY}`,
      'Accept': 'application/json'
    }
  }
);

const { experts, total } = await response.json();

// → { experts: [{ name, slug, topics, location, profileUrl }], total: 42 }

// Get a single expert profile
const profile = await fetch(
  'https://api.sisy.codes/v1/experts/linda-zavaa'
).then(r => r.json());
